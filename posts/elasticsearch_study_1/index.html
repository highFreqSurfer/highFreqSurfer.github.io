elastics的学习-1

November 20, 2024 · 高强度冲浪选手

创建mapping

put /good
{
    "mappings": {
        "properties": {
            "goodName": {
                "type": "text",
                "analyzer": "ik_smart"
            },
            "brand": {
                "type": "keyword", //精确词,不参与分词
                "index": false //是否创建倒排索引
            },
            "intro": {
                "type": "text", 
                "analyzer": "ik_smart" //分词器
            }
        }
    }
}

添加document

put /good/_doc/1
{
    "goodName": "我顶你妈的猪肺",
    "brand": "二师兄",
    "intro": "十分甚至九分的好吃"
}

全量更新document

put /good/_doc/1
{
    "goodName": "现吃现敲的猴脑",
    "brand": "大师兄",
    "intro": "特别的新鲜, 特别的OC"
}

单独更新document字段

put /good/_update/1
{
    "doc": {
        "goodName": "超级大飞柱"
    }
}

删除mapping

delete /good

删除document

delete /good/_doc/1

查询所有document

get /good/_search
{
    "query":{
        "match_all":{}
    }
}
